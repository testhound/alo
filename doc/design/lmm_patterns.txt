LMM patterns:
----------------------------------------------------------
        lmm r1:r1 r11
----------------------------------------------------------
 (set (reg:SI 1 r1)
        (mem:SI (reg:SI 11 r11) )
    

----------------------------------------------------------
	lmm r1:r1 4(r11)
----------------------------------------------------------
 (set (reg:SI 1 r1)
        (mem:SI (plus:SI (reg:SI 11 r11)
			 (const_int 4 [0x4])) )
    

----------------------------------------------------------
	lmm r1:r1 BASE2 foo
----------------------------------------------------------
 (set (reg:SI 1 r1)
        (mem:SI (plus:SI (reg:SI 101 BASE2)
			 (const:SI (symbol_ref:SI ("foo")))) )
    

----------------------------------------------------------
	lmm r1:r1 BASE3 foo
----------------------------------------------------------
 (set (reg:SI 1 r1)
        (mem:SI (plus:SI (reg:SI 102 BASE3)
			 (const:SI (symbol_ref:SI ("foo")))) )
